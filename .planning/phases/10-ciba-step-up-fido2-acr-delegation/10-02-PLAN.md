```rust
/// PAM -> Agent: initiate CIBA step-up authentication.
/// Agent spawns async CIBA poll loop and returns StepUpPending immediately.
#[serde(rename = "step_up")]
StepUp {
    username: String,
    command: String,
    hostname: String,
    /// "push" or "fido2" — determines acr_values sent to IdP
    method: String,
    /// From policy, default 120
    timeout_secs: u64,
},

/// PAM -> Agent: poll for step-up result.
/// Agent returns StepUpPending (still waiting), StepUpComplete, or StepUpTimedOut.
#[serde(rename = "step_up_result")]
StepUpResult {
    correlation_id: String,
},
```

Add to `AgentResponseData` enum (place StepUpPending, StepUpComplete, StepUpTimedOut BEFORE SessionAcknowledged — each has unique required fields for untagged serde discrimination):

```rust
/// Step-up initiated — PAM should poll with StepUpResult at poll_interval_secs.
StepUpPending {
    correlation_id: String,
    expires_in: u64,
    poll_interval_secs: u64,
},

/// Step-up completed successfully — ACR validated (if required).
StepUpComplete {
    acr: Option<String>,
    session_id: String,
},

/// Step-up failed or timed out.
StepUpTimedOut {
    reason: String,
    user_message: String,
},
```

```rust
/// Create a device flow client using endpoints from OIDC discovery.
///
/// This is the IdP-agnostic constructor. Reads `device_authorization_endpoint`
/// and `token_endpoint` from the discovery document. Returns an error if the
/// IdP does not advertise a device authorization endpoint.
///
/// Replaces `new()` for non-Keycloak deployments (STP-06).
pub fn from_discovery(
    discovery: &OidcDiscovery,
    client_id: &str,
    client_secret: Option<&str>,
) -> Result<Self, DeviceFlowError> {
    let device_endpoint = discovery.device_authorization_endpoint
        .as_deref()
        .ok_or_else(|| DeviceFlowError::ConfigError(
            "IdP does not advertise device_authorization_endpoint in OIDC discovery".to_string()
        ))?;

    Self::with_endpoints(device_endpoint, &discovery.token_endpoint, client_id, client_secret)
}
```
