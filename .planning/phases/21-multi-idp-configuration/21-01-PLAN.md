```rust
use std::collections::HashMap;
use std::sync::Arc;
use parking_lot::RwLock;

/// Thread-safe registry of per-issuer JWKS providers.
/// Invariant (MIDP-07): each issuer URL maps to an independent JwksProvider.
/// A fetch or refresh for issuer A NEVER touches the cache for issuer B.
pub struct IssuerJwksRegistry {
    providers: RwLock<HashMap<String, Arc<JwksProvider>>>,
}

impl IssuerJwksRegistry {
    pub fn new() -> Self {
        Self { providers: RwLock::new(HashMap::new()) }
    }

    /// Get or create the JwksProvider for the given issuer.
    /// Normalizes issuer URL (trims trailing slash) before lookup.
    pub fn get_or_init(&self, issuer: &str, ttl_secs: u64, timeout_secs: u64) -> Arc<JwksProvider> {
        let normalized = issuer.trim_end_matches('/');
        // Fast path: read lock
        {
            let read = self.providers.read();
            if let Some(p) = read.get(normalized) {
                return Arc::clone(p);
            }
        }
        // Write path: insert new provider
        let mut write = self.providers.write();
        write.entry(normalized.to_string())
            .or_insert_with(|| Arc::new(JwksProvider::with_timeouts(normalized, ttl_secs, timeout_secs)))
            .clone()
    }
}
```

2. Add `Default` impl for `IssuerJwksRegistry` (delegates to `new()`).

Anti-patterns to avoid:
- Do NOT use a global static for the registry. It will be owned by the auth routing struct (Plan 02).
- Do NOT acquire the write lock unconditionally — the read-first pattern avoids contention on the hot path.
