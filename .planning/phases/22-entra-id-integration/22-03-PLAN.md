    ```bash
    #!/usr/bin/env bash
    # Acquire an Entra ID access token via ROPC (Resource Owner Password Credentials).
    #
    # Usage: ./get-entra-token.sh
    #
    # Required environment variables:
    #   ENTRA_TENANT_ID     — Azure AD tenant ID (GUID)
    #   ENTRA_CLIENT_ID     — Application (client) ID (GUID)
    #   ENTRA_TEST_USER     — UPN of test user (e.g. ci-test@corp.example)
    #   ENTRA_TEST_PASSWORD — Password for test user
    #
    # Outputs the access token on stdout. Exits non-zero on failure.
    #
    # Security notes:
    #   - ROPC is incompatible with MFA. The test user MUST have MFA disabled
    #     or be excluded from Conditional Access MFA policies.
    #   - Do NOT use User.Read or other Graph scopes — they change the token
    #     audience to https://graph.microsoft.com (Pitfall 3, 22-RESEARCH.md).
    #   - Tokens are masked in CI via `::add-mask::`.
    #
    # References:
    #   - Microsoft Learn: OAuth 2.0 ROPC (https://learn.microsoft.com/en-us/entra/identity-platform/v2-oauth-ropc)
    set -euo pipefail

    : "${ENTRA_TENANT_ID:?ENTRA_TENANT_ID is required}"
    : "${ENTRA_CLIENT_ID:?ENTRA_CLIENT_ID is required}"
    : "${ENTRA_TEST_USER:?ENTRA_TEST_USER is required}"
    : "${ENTRA_TEST_PASSWORD:?ENTRA_TEST_PASSWORD is required}"

    TOKEN_ENDPOINT="https://login.microsoftonline.com/${ENTRA_TENANT_ID}/oauth2/v2.0/token"

    # Request scopes: openid + profile + email for user claims.
    # Do NOT add User.Read — it changes aud to Graph (see 22-RESEARCH.md Pitfall 3).
    SCOPE="openid profile email"

    RESPONSE=$(curl -sf -X POST "${TOKEN_ENDPOINT}" \
      -H "Content-Type: application/x-www-form-urlencoded" \
      -d "grant_type=password" \
      -d "client_id=${ENTRA_CLIENT_ID}" \
      -d "username=${ENTRA_TEST_USER}" \
      -d "password=${ENTRA_TEST_PASSWORD}" \
      -d "scope=${SCOPE}")

    # Extract access_token from JSON response.
    ACCESS_TOKEN=$(echo "${RESPONSE}" | jq -r '.access_token // empty')

    if [ -z "${ACCESS_TOKEN}" ]; then
      ERROR=$(echo "${RESPONSE}" | jq -r '.error_description // .error // "unknown error"')
      echo "ERROR: Failed to acquire Entra token: ${ERROR}" >&2
      exit 1
    fi

    echo "${ACCESS_TOKEN}"
    ```

    Make it executable: `chmod +x test/scripts/get-entra-token.sh`

  <files>.github/workflows/provider-tests.yml</files>
  <action>
    Read the existing `provider-tests.yml` first. Then make these changes:

    1. Update the `workflow_dispatch.inputs.provider.description` to include 'entra':

    2. Add the `entra` job between the `google` job and the `provider-summary` job.
       Follow the exact Auth0 pattern:

       ```yaml
       # ============================================================================
       # Entra ID Integration (Cloud, requires secrets)
       # ============================================================================
       entra:
         name: Entra ID Integration
         runs-on: ubuntu-latest
         if: |
           (github.event_name != 'pull_request' || github.event.pull_request.head.repo.full_name == github.repository) &&
           (github.event.inputs.provider == 'all' || github.event.inputs.provider == 'entra' || github.event.inputs.provider == '')
         steps:
           - uses: actions/checkout@v6

           - name: Check Entra Configuration
             id: check-entra
             run: |
               if [ -z "${{ secrets.ENTRA_TENANT_ID }}" ]; then
                 echo "ENTRA_TENANT_ID secret not configured - skipping Entra tests"
                 echo "skip=true" >> $GITHUB_OUTPUT
               else
                 echo "skip=false" >> $GITHUB_OUTPUT
               fi

           - name: Install Rust toolchain
             if: steps.check-entra.outputs.skip != 'true'
             uses: dtolnay/rust-toolchain@stable

           - name: Cache cargo registry
             if: steps.check-entra.outputs.skip != 'true'
             uses: actions/cache@v5
             with:
               path: |
                 ~/.cargo/registry
                 ~/.cargo/git
                 target
               key: ${{ runner.os }}-cargo-${{ hashFiles('**/Cargo.lock') }}

           - name: Install system dependencies
             if: steps.check-entra.outputs.skip != 'true'
             run: |
               sudo apt-get update
               sudo apt-get install -y libpam0g-dev libssl-dev pkg-config jq

           - name: Build
             if: steps.check-entra.outputs.skip != 'true'
             run: cargo build --release -p pam-unix-oidc

           - name: Verify Entra OIDC Discovery
             if: steps.check-entra.outputs.skip != 'true'
             env:
               ENTRA_TENANT_ID: ${{ secrets.ENTRA_TENANT_ID }}
             run: |
               echo "=== Entra OIDC Discovery ==="
               DISCOVERY_URL="https://login.microsoftonline.com/${ENTRA_TENANT_ID}/v2.0/.well-known/openid-configuration"
               curl -sf "${DISCOVERY_URL}" | jq .

               echo ""
               echo "=== Entra JWKS ==="
               JWKS_URI=$(curl -sf "${DISCOVERY_URL}" | jq -r '.jwks_uri')
               curl -sf "${JWKS_URI}" | jq '.keys | length' | xargs -I {} echo "Found {} keys"

               echo ""
               echo "=== Key Algorithms ==="
               curl -sf "${JWKS_URI}" | jq '.keys[].alg'

           - name: Acquire Entra Token (ROPC)
             if: steps.check-entra.outputs.skip != 'true'
             env:
               ENTRA_TENANT_ID: ${{ secrets.ENTRA_TENANT_ID }}
               ENTRA_CLIENT_ID: ${{ secrets.ENTRA_CLIENT_ID }}
               ENTRA_TEST_USER: ${{ secrets.ENTRA_TEST_USER }}
               ENTRA_TEST_PASSWORD: ${{ secrets.ENTRA_TEST_PASSWORD }}
             run: |
               chmod +x ./test/scripts/get-entra-token.sh
               TOKEN=$(./test/scripts/get-entra-token.sh)
               echo "::add-mask::${TOKEN}"
               echo "ENTRA_TOKEN=${TOKEN}" >> $GITHUB_ENV

               # Decode and display claims (without signature) for debugging
               echo ""
               echo "=== Token Claims (header) ==="
               echo "${TOKEN}" | cut -d. -f1 | base64 -d 2>/dev/null | jq . || true
               echo ""
               echo "=== Token Claims (payload) ==="
               echo "${TOKEN}" | cut -d. -f2 | base64 -d 2>/dev/null | jq '{iss, aud, sub, preferred_username, email, ver, iat, exp}' || true

           - name: Run Entra Integration Tests
             if: steps.check-entra.outputs.skip != 'true'
             env:
               ENTRA_TENANT_ID: ${{ secrets.ENTRA_TENANT_ID }}
               ENTRA_CLIENT_ID: ${{ secrets.ENTRA_CLIENT_ID }}
               ENTRA_TOKEN: ${{ env.ENTRA_TOKEN }}
             run: |
               cargo test --release -p pam-unix-oidc --test entra_integration \
                 -- --ignored --test-threads=1 2>&1

           - name: Entra Test Summary
             if: steps.check-entra.outputs.skip != 'true' && always()
             run: |
               echo "=== Entra Integration Test Results ==="
               echo "All Entra integration tests completed."
       ```

    3. Update the `provider-summary` job:
       - Add `entra` to the `needs` array: `needs: [keycloak, auth0, google, entra]`
       - Add Entra to the results output:
         ```yaml

         ```
       - Entra is optional (like Auth0) — do NOT add it to the required check

    4. Ensure the YAML is valid and properly indented.
